### 6.指针类型易错点

```c
//易错点
int main()
{
	int arr[] = { 1,2,3,4,5 };
	//易错点，这里是short*
	short* p = (short*)arr;
	int i = 0;

	for (i = 0; i < 4; i++)
	{
		*(p + i) = 0;//解引用时，一次只能访问2个字节，所以只把1，2改了
	}

	for (i = 0; i < 5; i++)
	{
		printf("%d ", arr[i]);
	}
	return 0;
}
```

### 7.%x

%x---打印16进制数

```c
int i;//全局变量未初始化，默认值为0
int main()
{
	i--; //-1
	if (i > sizeof(i))//i>sizeof(i)  int与unsigned int 进行比较，所以int要转换成unsigned int
		//-1---01111111111111111111111111111111---超级大
	{
		printf(">\n");
	}
	else
	{
		printf("<\n");
	}

	return 0;
}
```

### 10.n = n&(n-1)
